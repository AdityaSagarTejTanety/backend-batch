** We can export multiple components in a same file but not in a default manner. So, we need to import those multiple components using {}.

import React, { Component } from 'react'

export default class app extends Component {
  submitAction=()=>{
    alert()
  }
  handleChange=(event)=>{
    console.log(event.target.value)
  }
  render() {
    return (
      <div>
        <input onChange={this.handleChange} placeholder="Enter a name"/>
        <button onClick={this.submitAction}>reset</button>
        <button onClick={this.submitAction}>Submit</button>
      </div>
    )
  }
}
